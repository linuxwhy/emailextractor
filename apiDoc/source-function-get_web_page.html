 43: /**
 44:  * Get a file on the web.  The file may be an (X)HTML page, an image, etc.
 45:  * Return an associative array containing the page header, contents,
 46:  * and HTTP status code.
 47:  *
 48:  * Values in the returned array are as defined by the CURL curl_getinfo()
 49:  * function, and include:
 50:  *
 51:  *  "url"       the last effective URL after redirects
 52:  *  "http_code" the last error/status code
 53:  *  "content_type"  the content type from the header
 54:  *
 55:  * This function adds a few more:
 56:  *
 57:  *  "content"   the page content (text, image, etc.)
 58:  *  "errno"     the CURL error code
 59:  *  "errmsg"    the CURL error message
 60:  *
 61:  * On success, "errno" is 0, "http_code" is 200, and "content" has the
 62:  * web page.
 63:  *
 64:  * On an error with the URL, such as a redirect limit, or timeout,
 65:  * "errno" will be non-zero and "errmsg" will contain an error message.
 66:  * There other fields will be missing.
 67:  *
 68:  * On an error with the web site, such as a missing page, no permissions,
 69:  * or no service, "errno" will be 0, "http_code" will be the HTTP error
 70:  * code, and "content" will be missing.
 71:  *
 72:  * Parameters:
 73:  *  url     the URL of the page to get
 74:  *
 75:  * Return values:
 76:  *  An associative array containing the page text and error codes,
 77:  *  as described above.
 78:  *
 79:  * See also:
 80:  *  http://nadeausoftware.com/articles/2007/06/php_tip_how_get_web_page_using_curl
 81:  */
 82: function get_web_page( $url )
 83: {
 84:     $options = array(
 85:         CURLOPT_RETURNTRANSFER => true,     // return web page
 86:         CURLOPT_HEADER         => false,    // don't return headers
 87:         CURLOPT_FOLLOWLOCATION => true,     // follow redirects
 88:         CURLOPT_ENCODING       => "",       // handle compressed
 89:         CURLOPT_USERAGENT      => "spider", // who am i
 90:         CURLOPT_AUTOREFERER    => true,     // set referer on redirect
 91:         CURLOPT_CONNECTTIMEOUT => 120,      // timeout on connect
 92:         CURLOPT_TIMEOUT        => 120,      // timeout on response
 93:         CURLOPT_MAXREDIRS      => 10,       // stop after 10 redirects
 94:     );
 95: 
 96:     $ch      = curl_init( $url );
 97:     curl_setopt_array( $ch, $options );
 98:     $content = curl_exec( $ch );
 99:     $err     = curl_errno( $ch );
100:     $errmsg  = curl_error( $ch );
101:     $header  = curl_getinfo( $ch );
102:     curl_close( $ch );
103: 
104:     $header['errno']   = $err;
105:     $header['errmsg']  = $errmsg;
106:     $header['content'] = $content;
107:     return $header;
108: }
